SELECT DEPTNO,ROUND(AVG(SAL)) AS AVG_SAL FROM EMP GROUP BY DEPTNO;

SELECT DEPTNO,ROUND(AVG(SAL)) AS AVG_SAL FROM EMP GROUP BY DEPTNO HAVING ROUND(AVG(SAL))>2000;

SELECT DEPTNO,MAX(SAL) FROM EMP GROUP BY DEPTNO HAVING MAX(SAL)>2900;

SELECT ROUND(MAX(AVG(SAL))) AS MAX_AVG_SAL FROM EMP GROUP BY DEPTNO;

SELECT JOB,COUNT(JOB) AS NO_OF_PEOPLE FROM EMP GROUP BY JOB;

SELECT COUNT(JOB) AS MANAGERS FROM EMP WHERE JOB='MANAGER';

SELECT MGR,MIN(SAL) AS MIN_SAL FROM EMP WHERE MGR IS NOT NULL GROUP BY MGR;

SELECT ENAME,SAL FROM EMP WHERE SAL>(SELECT SAL FROM EMP WHERE EMPNO=7566);

SELECT ENAME,JOB FROM EMP WHERE JOB=(SELECT JOB FROM EMP WHERE EMPNO=7369);

SELECT ENAME,JOB,SAL FROM EMP WHERE SAL=(SELECT MIN(SAL) FROM EMP);

SELECT ENAME,EMPNO,SAL FROM EMP WHERE SAL>(SELECT AVG(SAL) FROM EMP);
